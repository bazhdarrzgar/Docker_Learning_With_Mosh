* Directory and find

```bash

# show what is inside directory
ls             or              ls name_directory

# show what is inside this directory with hidden file
ls -a          or              ls -a name_directory


# find something anything in that directory with hidden directory
find              or              find name_directory

# example
find /etc

===
.
.
.
===
 
# find with name this componenet you want with hidden component
find name_componenet

# find just directory
find -type d

# find just file
find -type f

# find any file with anything start with f
find -type f -name "f*"

# find something with caseinsensitive
find -type f -iname "F*"

# find any file in root directory that is python file
find / -type f -name "*.py"

# redirect the output to the file.txt
find / -type f -name "*.py" > python-files.txt

```
